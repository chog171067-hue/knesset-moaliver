if (event.httpMethod === 'OPTIONS') { return { statusCode: 200, headers, body: '' }; } if (event.httpMethod !== 'POST') { return { statusCode: 405, headers, body: 'Method Not Allowed' }; } try { const { tz } = JSON.parse(event.body); if (!tz) { return { statusCode: 400, headers, body: JSON.stringify({ allowed: false, error: 'תעודת זהות חסרה' }) }; } // בדיקת הרשאה (מאושר תמיד לצורך הבדיקות כרגע) const isAllowedInCommunity = true; if (!isAllowedInCommunity) { return { statusCode: 200, headers, body: JSON.stringify({ allowed: false, exists: false }) }; } // פנייה ל-API המרכזי של נטליפיי לבדיקת קיום החשבון const checkResult = await checkIfUserExistsInIdentity(tz); return { statusCode: 200, headers, body: JSON.stringify({ allowed: true, exists: checkResult.exists, debugMessage: checkResult.debugMessage || '' }) }; } catch (error) { console.error('Error in verify-tz:', error); return { statusCode: 200, headers, body: JSON.stringify({ allowed: true, exists: false, debugError: error.message }) }; } }; async function checkIfUserExistsInIdentity(tz) { const identityToken = process.env.NETLIFY_IDENTITY_ADMIN_TOKEN; const siteId = process.env.SITE_ID; // נטליפיי מספקת את מזהה האתר אוטומטית לשרת if (!identityToken) { return { exists: false, debugMessage: "NETLIFY_IDENTITY_ADMIN_TOKEN defines are missing" }; } if (!siteId) { return { exists: false, debugMessage: "SITE_ID environment variable is missing" }; } try { // פנייה ישירה ומאובטחת ל-API המרכזי של Netlify בעזרת האסימון הראשי שלך const response = await axios.get(`https://api.netlify.com/api/v1/sites/${siteId}/identity/users`, { headers: { Authorization: `Bearer ${identityToken}`, 'Content-Type': 'application/json' } }); const users = response.data.users || []; // חיפוש משתמש קיים עם תעודת הזהות הזו const existingUser = users.find(u => { return u.user_metadata && String(u.user_metadata.verified_tz) === String(tz); }); return { exists: !!existingUser }; } catch (err) { const errorDetails = err.response ? JSON.stringify(err.response.data) : err.message; console.error("Netlify Central API Error:", errorDetails); return { exists: false, debugMessage: `API Error: ${errorDetails}` }; } }
